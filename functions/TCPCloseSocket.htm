TCPCloseSocket

Closes a TCP socket.Ferme un socket TCP.

TCPCloseSocket ( socket )

ParametersParamètres

socket The socket identifier (SocketID) as returned by a TCPListen(),TCPConnect() or TCPAccept() functions.

Return ValueValeur de retour

Success:Succès : 1.Retourne 1 si la fenêtre existe.
Failure:Échec : 0 and sets the @error flag to non-zero.Retourne 0 et @error <> 0.
@error:@error : Windows API WSAGetError return value (see MSDN).Valeur retournée par l'API Windows WSAGetError (Consultez MSDN).

#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>

; Start First clicking on "1.Commencez d'abord par cliquer sur "1. Server"Serveur"

Example()

Func Example()
    TCPStartup() ; Start the TCP service.Démarre le service TCP.

    ; Register OnAutoItExit to be called when the script is closed.Inscrit OnAutoItExit qui sera appelée lorsque le script se fermera.
    OnAutoItExitRegister("OnAutoItExit")

    ; Assign Local variables the loopback IP Address and the Port.Assigne des variables locales avec l'adresse IP et le port.
    Local $sIPAddress = "127.0.0.1" ; This IP Address only works for testing on your own computer.Cette adresse IP ne fonctionne que pour les tests sur votre propre ordinateur.
    Local $iPort = 65432 ; Port used for the connection.Port utilisé pour la connexion.

    #Region GUI
    Local $sTitle = "TCP Start"
    Local $hGUI = GUICreate($sTitle, 250, 70)

    Local $idBtnServer = GUICtrlCreateButton("1.Retourne 1 si la fenêtre existe. Server", 65, 10, 130, 22)

    Local $idBtnClient = GUICtrlCreateButton("2. Client", 65, 40, 130, 22)

    GUISetState(@SW_SHOW, $hGUI)

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
            Case $idBtnServer
                WinSetTitle($sTitle, "", "TCP Server started")
                GUICtrlSetState($idBtnClient, $GUI_HIDE)
                GUICtrlSetState($idBtnServer, $GUI_DISABLE)
                If Not MyTCP_Server($sIPAddress, $iPort) Then ExitLoop
            Case $idBtnClient
                WinSetTitle($sTitle, "", "TCP Client started")
                GUICtrlSetState($idBtnServer, $GUI_HIDE)
                GUICtrlSetState($idBtnClient, $GUI_DISABLE)
                If Not MyTCP_Client($sIPAddress, $iPort) Then ExitLoop
        EndSwitch

        Sleep(10)
    WEnd

    #EndRegion GUI
EndFunc   ;==>Example

Func MyTCP_Client($sIPAddress, $iPort)
    ; Assign a Local variable the socket and connect to a listening socket with the IP Address and Port specified.Assigne une variable locale avec le socket et se connecte à un socket d'écoute avec l'adresse IP et le port spécifié.
    Local $iSocket = TCPConnect($sIPAddress, $iPort)
    Local $iError = 0

    ; If an error occurred display the error code and return False.Si une erreur s'est produite, affiche le code d'erreur et retourne.
    If @error Then
        ; The server is probably offline/port is not opened on the server.Le serveur est probablement hors ligne ou le port n'est pas ouvert sur le serveur.
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Client:" & @CRLF & "Could not connect, Error code: " & $iError)
        Return False
    EndIf

    ; Send the string "tata" to the server.Envoie la chaîne "tata" au serveur.
    TCPSend($iSocket, "tata")

    ; If an error occurred display the error code and return False.Si une erreur s'est produite, affiche le code d'erreur et retourne.
    If @error Then
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Client:" & @CRLF & "Could not send the data, Error code: " & $iError)
        Return False
    EndIf

    ; Close the socket.Ferme le socket.
    TCPCloseSocket($iSocket)
EndFunc   ;==>MyTCP_Client

Func MyTCP_Server($sIPAddress, $iPort)
    ; Assign a Local variable the socket and bind to the IP Address and Port specified with a maximum of 100 pending connexions.Assigne une variable locale avec le socket et se connecte à l'adresse IP et au port spécifié avec un maximum de 100 connexions en attente.
    Local $iListenSocket = TCPListen($sIPAddress, $iPort, 100)
    Local $iError = 0

    If @error Then
        ; Someone is probably already listening on this IP Address and Port (script already running?).Quelqu'un est probablement déjà à l'écoute sur cette adresse IP et ce port (script déjà en cours d'exécution?).
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Server:" & @CRLF & "Could not listen, Error code: " & $iError)
        Return False
    EndIf

    ; Assign a Local variable to be used by the Client socket.Assigne une variable locale pour être utilisé par le socket client.
    Local $iSocket = 0

    Do ; Wait for someone to connect (Unlimited).Attend que quelqu'un se connecte (illimité).
        ; Accept incomming connexions if present (Socket to close when finished; one socket per client).Accepte des connexions entrantes si présent (Socket fermé une fois terminé, un socket par client).
        $iSocket = TCPAccept($iListenSocket)

        ; If an error occurred display the error code and return False.Si une erreur s'est produite, affiche le code d'erreur et retourne.
        If @error Then
            $iError = @error
            MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Server:" & @CRLF & "Could not accept the incoming connection, Error code: " & $iError)
            Return False
        EndIf

        If GUIGetMsg() = $GUI_EVENT_CLOSE Then Return False
    Until $iSocket <> -1 ;if different from -1 a client is connected.si il est différent de -1 le client est connecté.

    ; Close the Listening socket to allow afterward binds.Ferme le socket d'écoute pour permettre pour permettre une connexion ultérieure.
    TCPCloseSocket($iListenSocket)

    ; Assign a Local variable the data received.Assigne une variable locale avec les données reçues.
    Local $sReceived = TCPRecv($iSocket, 4) ;we're waiting for the string "tata" OR "toto" (example script TCPRecv):nous attendons la chaîne "tata" OU "toto" (exemple de script TCPRecv): 4 bytes length.4 octets de longueur.

    ; Notes: If you don't know how much length will be the data,Si vous ne savez pas de quelle longueur seront les données,
    ; use e.g:utilisez, par exemple: 2048 for maxlen parameter and call the function until the it returns nothing/error.2048 pour paramètre maxlen et appelez la fonction jusqu'à ce qu'elle ne retourne rien ou erreur.

    ; Display the string received.Affiche la chaîne reçue.
    MsgBox($MB_SYSTEMMODAL, "", "Server:" & @CRLF & "Received: " & $sReceived)

    ; Close the socket.Ferme le socket.
    TCPCloseSocket($iSocket)
EndFunc   ;==>MyTCP_Server

Func OnAutoItExit()
    TCPShutdown() ; Close the TCP service.Ferme le service TCP.
EndFunc   ;==>OnAutoItExit
