#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Assign a Local variable an array containing the numbers.Assigne une variable locale avec tableau contenant les chiffres.
    Local $aNumber[8] = [4.8, 4.5, 4.3, 4, -4.8, -4.5, -4.3, -4]

    ; Assign a Local variable a string which will contain the results.Assigne une variable locale avec chaîne qui contiendra les résultats.
    Local $sResults = ""

    ; Loop through the array: calculate the ceiling and format the result.Boucle dans le tableau pour calculer l'arrondi supérieur et formater le résultat.
    For $i = 0 To 7
        $sResults &= "Ceiling(" & $aNumber[$i] & ") = " & Ceiling($aNumber[$i]) & @CRLF & ($i = 3 ? @CRLF : "")
    Next

    ; Display the results.Affiche les résultats.
    MsgBox($MB_SYSTEMMODAL, "", $sResults)
EndFunc   ;==>Example
