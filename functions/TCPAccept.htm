TCPAccept

Permits an incoming connection attempt on a socket.Permet une tentative de connexion entrante sur un socket.

TCPAccept ( mainsocket )

ParametersParamètres

mainsocket The main socket identifier (SocketID) as returned by a TCPListen() function.

Return ValueValeur de retour

Success:Succès : the connected socket identifier.Retourne l'identifiant du socket connecté.
Failure:Échec : -1 and sets the @error flag to non-zero.-1 et met @error <> 0.
@error:@error : -2 not connected.
Windows API WSAGetError return value (see MSDN).Valeur retournée par l'API Windows WSAGetError (Consultez MSDN).

#include <MsgBoxConstants.au3>

Func Example()
    TCPStartup() ; Start the TCP service.Démarre le service TCP.

    ; Register OnAutoItExit to be called when the script is closed.Inscrit OnAutoItExit qui sera appelée lorsque le script se fermera.
    OnAutoItExitRegister("OnAutoItExit")

    ; Assign Local variables the loopback IP Address and the Port.Assigne des variables locales avec l'adresse IP et le port.
    Local $sIPAddress = "127.0.0.1" ; This IP Address only works for testing on your own computer.Cette adresse IP ne fonctionne que pour les tests sur votre propre ordinateur.
    Local $iPort = 65432 ; Port used for the connection.Port utilisé pour la connexion.

    ; Bind to the IP Address and Port specified with a maximum of 100 pending connexionsSe connecte à l'adresse IP sur le port spécifié avec un maximum de 100 connexions en attente
    ;(Take a look at the example of this function for further details).(Regardez l'exemple de cette fonction pour plus de détails).
    Local $iListenSocket = TCPListen($sIPAddress, $iPort, 100)
    Local $iError = 0

    ; If an error occurred display the error code and return False.Si une erreur s'est produite, affiche le code d'erreur et retourne.
    If @error Then
        ; Someone is probably already listening on this IP Address and Port (script already running?).Quelqu'un est probablement déjà à l'écoute sur cette adresse IP et ce port (script déjà en cours d'exécution?).
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Could not listen, Error code: " & $iError)
        Return False
    EndIf

    ; Assign Local variable to be used by Listening and Client sockets.Affecte la variable locale pour l'écoute et sockets clients.
    Local $iSocket = 0
    Do ; Wait for someone to connect (Unlimited).Attend que quelqu'un se connecte (illimité).
        ; Accept incomming connexions if present (Socket to close when finished; one socket per client).Accepte des connexions entrantes si présent (Socket fermé une fois terminé, un socket par client).
        $iSocket = TCPAccept($iListenSocket)

        ; If an error occurred display the error code and return False.Si une erreur s'est produite, affiche le code d'erreur et retourne.
        If @error Then
            $iError = @error
            MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Could not accept the incoming connection, Error code: " & $iError)
            Return False
        EndIf
    Until $iSocket <> -1 ;if different from -1 a client is connected.si il est différent de -1 le client est connecté.

    ; Close the Listening socket to allow afterward binds.Ferme le socket d'écoute pour permettre pour permettre une connexion ultérieure.
    TCPCloseSocket($iListenSocket)

    MsgBox($MB_SYSTEMMODAL, "", "Client Connected.")

    ; Close the socket.Ferme le socket.
    TCPCloseSocket($iSocket)
EndFunc   ;==>Example

Func OnAutoItExit()
    TCPShutdown() ; Close the TCP service.Ferme le service TCP.
EndFunc   ;==>OnAutoItExit

Func Example()
    TCPStartup()

    OnAutoItExitRegister("OnAutoItExit")

    Local $sIPAddress = "127.0.0.1"
    Local $iPort = 65432

    Local $iListenSocket = TCPListen($sIPAddress, $iPort, 100)
    Local $iError = 0

    If @error Then
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Could not listen, Error code: " & $iError)
        Return False
    EndIf

    Local $iSocket = 0
    Do
        $iSocket = TCPAccept($iListenSocket)

        If @error Then
            $iError = @error
            MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Could not accept the incoming connection, Error code: " & $iError)
            Return False
        EndIf
    Until $iSocket <> -1

    TCPCloseSocket($iListenSocket)

    ; Retrieve the IP Address associated with the accepted socket and assign it to a Local variable.Récupére l'adresse IP associée au socket acceptée et l'assigne à une variable locale.
    Local $sClientIPAddress = SocketToIP($iSocket)

    ; Note:Note : The above function does NOT work with the Listen socket, you can also use it with the socket returned by the TCPConnect function.La fonction ci-dessus ne fonctionne pas avec le socket Listen, vous pouvez également l'utiliser avec le socket renvoyée par la fonction TCPConnect.

    ; Display the sucessful message with the client IP Address.Affiche un message de réussite avec l'adresse IP du client.
    MsgBox($MB_SYSTEMMODAL, "", "Client Connected, IP Address: " & $sClientIPAddress)

    TCPCloseSocket($iSocket)
EndFunc   ;==>Example

Func SocketToIP($iSocket)
    Local $tSockAddr = 0, $aRet = 0
    $tSockAddr = DllStructCreate("short;ushort;uint;char[8]")
    $aRet = DllCall("Ws2_32.dll", "int", "getpeername", "int", $iSocket, "struct*", $tSockAddr, "int*", DllStructGetSize($tSockAddr))
    If Not @error And $aRet[0] = 0 Then
        $aRet = DllCall("Ws2_32.dll", "str", "inet_ntoa", "int", DllStructGetData($tSockAddr, 3))
        If Not @error Then Return $aRet[0]
    EndIf
    Return 0
EndFunc   ;==>SocketToIP

Func OnAutoItExit()
