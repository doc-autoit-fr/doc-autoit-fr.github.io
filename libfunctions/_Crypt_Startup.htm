Failure:Échec : False and sets the @error flag to non-zero.False et met @error <> 0.

ExampleExemple

#include <Crypt.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    _Crypt_Startup() ; To optimize performance start the crypt library, though the same results will be shown if it isn't.Pour optimiser les performances exécuter la bibliothèque crypt, même si ce sont les mêmes résultats affichés dans le cas contraire.

    Local Const $sUserKey = "CryptPassword" ; Declare a password string to decrypt/encrypt the data.Déclare un de mot de passe pour déchiffrer/chiffrer les données.
    Local $sData = "..upon a time there was a language without any standardized..... il était une fois un langage sans aucune norme..." ; Data that will be hashed.Données qui seront hachées.

    Local $sOutput = "The following results show the supported algorithms for encrypting the data.Les résultats suivants montrent les algorithmes pris en charge pour chiffrer les données." & @CRLF & @CRLF & _
            "Text:Texte : " & $sData & @CRLF & _
            "3DES:3DES : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_3DES) & @CRLF & @CRLF & _
            "AES (128bit):AES (128bit) : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_AES_128) & @CRLF & @CRLF & _
            "AES (192bit):AES (192bit) : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_AES_192) & @CRLF & @CRLF & _
            "AES (256bit):AES (256bit) : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_AES_256) & @CRLF & @CRLF & _
            "DES:DES : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_DES) & @CRLF & @CRLF & _
            "RC2:RC2 : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_RC2) & @CRLF & @CRLF & _
            "RC4:RC4 : " & _Crypt_EncryptData($sData, $sUserKey, $CALG_RC4)

    MsgBox($MB_SYSTEMMODAL, "Supported algorithmsAlgorithmes pris en charge", $sOutput)

    _Crypt_Shutdown() ; Shutdown the crypt library.Arrête la bibliothèque crypt.
EndFunc   ;==>Example
