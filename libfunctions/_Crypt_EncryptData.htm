_Crypt_EncryptData

Encrypts data using the supplied keyChiffre les données à l'aide de la clé fournie

#include <Crypt.au3>
_Crypt_EncryptData ( $vData, $vCryptKey, $iAlgID [, $bFinal = True] )

ParametersParamètres

$vData Data to encrypt/decryptDonnées à chiffrer/déchiffrer
$vCryptKey Password or handle to a key if the CALG_USERKEY flag is specifiedMot de passe ou descripteur (handle)'d'une clé si l'indicateur CALG_USERKEY est spécifié
$iAlgID The algorithm to useL'algorithme à utiliser
$bFinal [optional] False if this is only a segment of the full dataFalse si c'est seulement un segment de données complet

Return ValueValeur de retour

Success: a binary string containing encrypted data.une chaîne binaire contenant les données chiffrées.
Failure: -1 and sets the @error flag to non-zero.-1 et met @error <> 0.
@error:@error : 100+ - Cannot create key100+ - Ne peut créer la clé
20 - Failed to determine buffer20 - Impossible de déterminer la mémoire tampon
30 - Failed to encrypt data30 - Impossible de chiffrer les données

#include <ComboConstants.au3>

#include <MsgBoxConstants.au3>

#include <WindowsConstants.au3>

Global $g_hKey = -1, $g_idInputEdit = -1, $g_idOutputEdit = -1

Example()

Func Example()
    Local $hGUI = GUICreate("Realtime EncryptionChiffrement en temps réel", 400, 320)
    $g_idInputEdit = GUICtrlCreateEdit("", 0, 0, 400, 150, $ES_WANTRETURN)
    $g_idOutputEdit = GUICtrlCreateEdit("", 0, 150, 400, 150, $ES_READONLY)
    Local $idCombo = GUICtrlCreateCombo("", 0, 300, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "3DES|AES (128bit)|AES (192bit)|AES (256bit)|DES|RC2|RC4", "RC4")
    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")
    GUISetState(@SW_SHOW, $hGUI)

    _Crypt_Startup() ; To optimize performance start the crypt library.Exécute la bibliothèque crypt pour optimiser les performances.

    Local $iAlgorithm = $CALG_RC4
    $g_hKey = _Crypt_DeriveKey("CryptPassword", $iAlgorithm) ; Declare a password string and algorithm to create a cryptographic key.Déclare un mot de passe sous forme de châine et l'algorithme pour créer une clé de chiffrement.

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                Exit

            Case $idCombo ; Check when the combobox is selected and retrieve the correct algorithm.Vérifie si le combobox est sélectionné et récupère le bon algorithme.
                Switch GUICtrlRead($idCombo) ; Read the combobox selection.Lit la sélection du combobox.
                    Case "3DES"
                        $iAlgorithm = $CALG_3DES

                    Case "AES (128bit)"
                        $iAlgorithm = $CALG_AES_128

                    Case "AES (192bit)"
                        $iAlgorithm = $CALG_AES_192

                    Case "AES (256bit)"
                        $iAlgorithm = $CALG_AES_256

                    Case "DES"
                        $iAlgorithm = $CALG_DES

                    Case "RC2"
                        $iAlgorithm = $CALG_RC2

                    Case "RC4"
                        $iAlgorithm = $CALG_RC4

                EndSwitch

                _Crypt_DestroyKey($g_hKey) ; Destroy the cryptographic key.Détruit la clé de chiffrement.
                $g_hKey = _Crypt_DeriveKey("CryptPassword", $iAlgorithm) ; Re-declare a password string and algorithm to create a new cryptographic key.Re-déclare un mot de passe sous forme de chaîne et l'algorithme pour créer une nouvelle clé de chiffrement.

                Local $sRead = GUICtrlRead($g_idInputEdit)
                If StringStripWS($sRead, $STR_STRIPALL) <> "" Then ; Check there is text available to encrypt.Vérifie qu'il y a du texte disponible à chiffrer.
                    Local $bEncrypted = _Crypt_EncryptData($sRead, $g_hKey, $CALG_USERKEY) ; Encrypt the text with the new cryptographic key.Chiffre le texte avec la nouvelle clé cryptographique.
                    GUICtrlSetData($g_idOutputEdit, $bEncrypted) ; Set the output box with the encrypted text.Définit le contenu de la boîte de dialogue de sortie avec le texte chiffré.
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Delete the previous GUI and all controls.Supprime la GUI précédente et tous ses contrôles.
    _Crypt_DestroyKey($g_hKey) ; Destroy the cryptographic key.Détruit la clé de chiffrement.
    _Crypt_Shutdown() ; Shutdown the crypt library.Arrête la bibliothèque crypt.
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $lParam

    Switch _WinAPI_LoWord($wParam)
        Case $g_idInputEdit
            Switch _WinAPI_HiWord($wParam)
                Case $EN_CHANGE
                    Local $bEncrypted = _Crypt_EncryptData(GUICtrlRead($g_idInputEdit), $g_hKey, $CALG_USERKEY) ; Encrypt the text with the cryptographic key.Chiffre le texte avec la clé de chiffrement.
                    GUICtrlSetData($g_idOutputEdit, $bEncrypted) ; Set the output box with the encrypted text.Définit le contenu de la boîte de dialogue de sortie avec le texte chiffré.
            EndSwitch
    EndSwitch
EndFunc   ;==>WM_COMMAND

Func Example()
    ; Encrypt text using a generic password.Chiffre le texte à l'aide d'un mot de passe générique.
    Local $sEncrypted = StringEncrypt(True, 'Encrypt this data.Chiffre ces données.', 'securepasswordmotdepassesécurisé')

    ; Display the encrypted text.Affiche le texte chiffré.
    MsgBox($MB_SYSTEMMODAL, '', $sEncrypted)

    ; Decrypt the encrypted text using the generic password.Déchiffre le texte chiffré en utilisant le mot de passe générique.
    Local $sDecrypted = StringEncrypt(False, $sEncrypted, 'securepasswordmotdepassesécurisé')

    ; Display the decrypted text.Affiche le texte déchiffré.
    MsgBox($MB_SYSTEMMODAL, '', $sDecrypted)
EndFunc   ;==>Example

Func StringEncrypt($bEncrypt, $sData, $sPassword)
    _Crypt_Startup() ; Start the Crypt library.Exécute la bibliothèque Crypt.
    Local $sReturn = ''
    If $bEncrypt Then ; If the flag is set to True then encrypt, otherwise decrypt.Si l'indicateur est défini à True alors chiffrer, sinon déchiffrer.
        $sReturn = _Crypt_EncryptData($sData, $sPassword, $CALG_RC4)
    Else
        $sReturn = BinaryToString(_Crypt_DecryptData($sData, $sPassword, $CALG_RC4))
    EndIf
    _Crypt_Shutdown() ; Shutdown the Crypt library.Arrête la bibliothèque Crypt.
    Return $sReturn
EndFunc   ;==>StringEncrypt
