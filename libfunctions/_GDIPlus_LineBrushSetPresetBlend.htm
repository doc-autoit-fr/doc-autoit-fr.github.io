_GDIPlus_LineBrushSetPresetBlend

Sets the colors to be interpolated for a linear gradient brush and their corresponding blend positionsPour un pinceau à dégradé linéaire existant, définit les couleurs à mélanger et les positions correspondantes

#include <GDIPlus.au3>
_GDIPlus_LineBrushSetPresetBlend ( $hLineGradientBrush, $aInterpolations )

ParametersParamètres

$hLineGradientBrush Pointer to a LinearGradientBrush objectHandle de l'objet LinearGradientBrush
$aInterpolations Array of blend colors and blend positions:Tableau de couleurs à mélanger et positions correspondantes:
    [0][0] - Number of blend colors and blend positions, must be at least 2
    [1][0] - Color 1
    [1][1] - Position 1
    [2][0] - Color 2
    [2][1] - Position 2
    [n][0] - Color n
    [n][1] - Position n

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False and sets the @error flag to non-zero, @extended may contain GPSTATUS error code ($GPID_ERR*).Retourne False et @error <> 0, @extended contient le code erreur GPSTATUS($GPID_ERR*).

ExampleExemple

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    _GDIPlus_Startup() ;initialize GDI+Initialize GDI+
    Local Const $iWidth = 600, $iHeight = 300, $iBgColor = 0x303030 ;$iBgColor format RRGGBB$iBGColor est au format RRGGBB

    Local $hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;create a test GUIcréer une interface de test
    GUISetBkColor($iBgColor, $hGUI) ;set GUI background colorDéfinit la couleur de fond de la GUI
    GUISetState(@SW_SHOW)

    Local $hGraphics = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;create a graphics object from a window handleCrée un objet graphique à partir du handle de fenêtre
    _GDIPlus_GraphicsSetSmoothingMode($hGraphics, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;sets the graphics object rendering quality (antialiasing)Définit l'objet graphique en qualité de rendu antialiasing

    Local $hBrush = _GDIPlus_LineBrushCreate(0, 0, 580, 280, 0, 0, 1) ;create line brushCrée un pinceau

    Local $aInterpolations[5][2] ;define the interpolated colors and positionsDéfinit les couleurs interpolées et les emplacements
    $aInterpolations[0][0] = 4
    $aInterpolations[1][0] = 0xFFFF0000 ;RedRouge
    $aInterpolations[1][1] = 0 ;0% from the left0 % à partir de la gauche
    $aInterpolations[2][0] = 0xFF00FF00 ;GreenVert
    $aInterpolations[2][1] = 0.3 ;To 30% from the leftJusqu'à 30 % à partir de la gauche
    $aInterpolations[3][0] = 0xFF0000FF ;BlueBleu
    $aInterpolations[3][1] = 0.7 ;To 70% from the leftJusqu'à 70% à partir de la gauche
    $aInterpolations[4][0] = 0xFFFFFF00 ;YellowJaune
    $aInterpolations[4][1] = 1 ;To 100% from the leftJusqu'à 100% à partir de la gauche

    _GDIPlus_LineBrushSetPresetBlend($hBrush, $aInterpolations) ;set the linear gradient brush colors and positionDéfinir les couleurs et les emplacements de pinceau de dégradé linéaire

    _GDIPlus_GraphicsFillRect($hGraphics, 10, 10, 580, 280, $hBrush) ;draw bitmap to screenDessine le bitmap à l'écran

    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ;cleanup GDI+ resourcesNettoie les ressources de GDI+
    _GDIPlus_BrushDispose($hBrush)
    _GDIPlus_GraphicsDispose($hGraphics)
    _GDIPlus_Shutdown()
    GUIDelete($hGUI)
EndFunc   ;==>Example
