_GDIPlus_GraphicsFillClosedCurve2

Creates a closed cardinal spline from an array of points and uses a brush to fill the interior of the splineCrée une spline cardinale fermée à partir d'un tableau de points et utilise un pinceau pour remplir l'intérieur de la spline

#include <GDIPlus.au3>
_GDIPlus_GraphicsFillClosedCurve2 ( $hGraphics, $aPoints, $nTension [, $hBrush = 0 [, $iFillMode = 0]] )

ParametersParamètres

$hGraphics Pointer to a Graphics objectLe handle de l'objet Graphics
$aPoints Array of points that specify the coordinates of the closed cardinal spline:Tableau de points qui spécifie les coordonnées de la spline cardinale fermée:
    [0][0] - Number of points
    [1][0] - Point 1 X position
    [1][1] - Point 1 Y position
    [2][0] - Point 2 X position
    [2][1] - Point 2 Y position
    [n][0] - Point n X position
    [n][1] - Point n Y position
$nTension Number that specifies how tightly the spline bends as it passes through the pointsNombre qui spécifie comment la spline sera tendue sur les points
$hBrush [optional][optionnel][optionnel] Handle to a brush object that is used to fill the cardinal spline.Handle de l'objet Brush (pinceau) qui est utilisé pour remplir la spline cardinale. If 0, a black brush will be used.Si 0, un pinceau noir sera utilisée.
$iFillMode [optional][optionnel][optionnel] Fill mode of the interior of the spline:Mode de remplissage de l'intérieur de la spline:
    0 - The areas are filled according to the even-odd parity rule
    1 - The areas are filled according to the nonzero winding rule

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False and sets the @error flag to non-zero, @extended may contain GPSTATUS error code ($GPID_ERR*).Retourne False et @error <> 0, @extended contient le code erreur GPSTATUS($GPID_ERR*).

ExampleExemple

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Global $g_hGUI, $g_hGfxCtxt, $g_hBitmap, $g_hGraphics, $g_hBrush

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ;initialize GDI+Initialize GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor format RRGGBB$iBGColor est de la forme RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;create a test GUIcréer une interface de test
    GUISetBkColor($iBgColor, $g_hGUI) ;set GUI background colorDéfinit la couleur de fond de la GUI
    GUISetState(@SW_SHOW)

    ;create buffered graphics frame set for smoother gfx object movementsCrée un buffer graphique pour lisser les objets gfx en mouvement
    $g_hGraphics = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;create a graphics object from a window handleCrée un objet graphique à partir du handle de fenêtre
    $g_hBitmap = _GDIPlus_BitmapCreateFromGraphics($iWidth, $iHeight, $g_hGraphics)
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBitmap)
    _GDIPlus_GraphicsSetSmoothingMode($g_hGfxCtxt, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;sets the graphics object rendering quality (antialiasing)Définit l'objet graphique en qualité de rendu antialiasing
    $g_hBrush = _GDIPlus_BrushCreateSolid(0xFF80FF60)

    Local $aPoints[11][4], $x, $y
    $aPoints[0][0] = 10

    For $y = 0 To 1
        For $x = 1 To 5
            $aPoints[$y * 5 + $x][0] = 100 + 300 * $y + 50 ;x coordinate of the pointcoordonnée x du point
            $aPoints[$y * 5 + $x][1] = 150 + $x * 50 ;y coordinate of the pointcoordonnée y du point
            $aPoints[$y * 5 + $x][2] = Random(-2, 2) ;x vector of the pointAbscisse du vecteur associé au point
            $aPoints[$y * 5 + $x][3] = Random(-2, 2) ;y vector of the pointOrdonnée du vecteur associé au point
        Next
    Next

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ;clear bitmap for repaintEfface bitmap pour le repeindre
        _GDIPlus_GraphicsFillClosedCurve2($g_hGfxCtxt, $aPoints, 0.75, $g_hBrush) ;draw closed curveTrace la courbe fermée
        _GDIPlus_GraphicsDrawImage($g_hGraphics, $g_hBitmap, 0, 0) ;copy bitmap to graphic handle (GUI)Copie bitmap dans le handle Graphics (GUI)
        For $y = 1 To $aPoints[0][0]
            $aPoints[$y][0] += $aPoints[$y][2] ;calculate new x positioncalculer nouvelle position x
            If $aPoints[$y][0] < 0 Or $aPoints[$y][0] > $iWidth Then $aPoints[$y][2] *= -1 ;if vertical border is reached invert x vectorSi la bordure verticale est atteinte l'abscisse du vecteur change de signe
            $aPoints[$y][1] += $aPoints[$y][3] ;calculate new y positionCalculer la nouvelle position y
            If $aPoints[$y][1] < 0 Or $aPoints[$y][1] > $iHeight Then $aPoints[$y][3] *= -1 ;if horizontal border is reached invert y vectorSi la bordure horizontale est atteinte l'ordonnée du vecteur change de signe
        Next
    Until Not Sleep(30) ;sleep 30 ms to avoid high cpu usagePause 30 ms pour éviter une utilisation élevée du processeur
EndFunc   ;==>Example

Func _Exit()
    ;cleanup GDI+ resourcesNettoie les ressources de GDI+
    _GDIPlus_BrushDispose($g_hBrush)
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_GraphicsDispose($g_hGraphics)
    _GDIPlus_BitmapDispose($g_hBitmap)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
