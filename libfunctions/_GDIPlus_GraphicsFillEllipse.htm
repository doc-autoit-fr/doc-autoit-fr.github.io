_GDIPlus_GraphicsFillEllipse

Fill an ellipseRemplit une ellipse

#include <GDIPlus.au3>
_GDIPlus_GraphicsFillEllipse ( $hGraphics, $nX, $nY, $nWidth, $nHeight [, $hBrush = 0] )

ParametersParamètre

$hGraphics Handle to a Graphics objectHandle de l'objet Graphics
$nX The X coordinate of the upper left corner of the rectangle that bounds the ellipseLa coordonnée X du coin supérieur gauche du rectangle qui délimite l'ellipse
$nY The Y coordinate of the upper left corner of the rectangle that bounds the ellipseLa coordonnée Y du coin supérieur gauche du rectangle qui délimite l'ellipse
$nWidth The width of the rectangle that bounds the ellipseLa largeur du rectangle qui délimite l'ellipse
$nHeight The height of the rectangle that bounds the ellipseLa hauteur du rectangle qui délimite l'ellipse
$hBrush [optional][optional] False - i, v, x ... Handle to a brush object that is used to fill the ellipse.Handle de l'objet Brush (pinceau) qui est utilisé pour remplir l'ellipse. If 0, a black brush will be used.Si 0, un pinceau noir sera utilisée.

Return ValueRemarque

Success: True.Vrai.
Failure: False and sets the @error flag to non-zero, @extended may contain GPSTATUS error code ($GPID_ERR*).Retourne False et @error <> 0, @extended contient le code erreur GPSTATUS($GPID_ERR*).

Example 1Exemple 1

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    Local $hGUI, $hGraphic

    ; Create GUICrée la GUI
    $hGUI = GUICreate("GDI+", 400, 300)
    GUISetState(@SW_SHOW)

    ; Draw an ellipseDessine une ellipse
    _GDIPlus_Startup()
    $hGraphic = _GDIPlus_GraphicsCreateFromHWND($hGUI)
    _GDIPlus_GraphicsFillEllipse($hGraphic, 130, 100, 140, 70)

    ; Loop until the user exits.Création d'un objet chemin
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; Clean up resourcesLibère les ressources
    _GDIPlus_GraphicsDispose($hGraphic)
    _GDIPlus_Shutdown()
EndFunc   ;==>Example

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    _GDIPlus_Startup() ;initialize GDI+Initialize GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBgColor format RRGGBB$iBGColor est au format RRGGBB

    Local $hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;create a test GUIcréer une interface de test
    GUISetBkColor($iBgColor, $hGUI) ;set GUI background colorDéfinit la couleur de fond de la GUI
    GUISetState(@SW_SHOW)

    Local $hGraphics = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;create a graphics object from a window handleCrée un objet graphique à partir du handle de fenêtre
    _GDIPlus_GraphicsSetSmoothingMode($hGraphics, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;sets the graphics object rendering quality (antialiasing)Définit l'objet graphique en qualité de rendu antialiasing
    Local $hBrush = _GDIPlus_BrushCreateSolid(0xFFE0A0FF) ;color format AARRGGBB (hex)Le format de couleur est AARRGGBB (hex)

    _GDIPlus_GraphicsFillEllipse($hGraphics, 10.5, 50.1, 580.25, 500.75, $hBrush)

    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ;cleanup GDI+ resourcesNettoie les ressources de GDI+
    _GDIPlus_BrushDispose($hBrush)
    _GDIPlus_GraphicsDispose($hGraphics)
    _GDIPlus_Shutdown()
    GUIDelete($hGUI)
EndFunc   ;==>Example
