_GDIPlus_GraphicsDrawImageRect

Draws an image at a specified locationDesssine une image à une position spécifiée

#include <GDIPlus.au3>
_GDIPlus_GraphicsDrawImageRect ( $hGraphics, $hImage, $nX, $nY, $nW, $nH )

ParametersParamètres

$hGraphics Handle to a Graphics objectHandle de l'objet Graphics
$hImage Handle to an Image objectHandle de l'objet l'image
$nX The X coordinate of the upper left corner of the rendered imageLa coordonnée X du coin supérieur gauche de l'image
$nY The Y coordinate of the upper left corner of the rendered imageLa coordonnée Y du coin supérieur gauche de l'image
$nW Specifies the width of the destination rectangle at which to draw the imageSpécifie la largeur du rectangle de destination dans lequel l'image sera dessinée
$nH Specifies the height of the destination rectangle at which to draw the imageSpécifie la hauteur du rectangle de destination dans lequel l'image sera dessinée

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False and sets the @error flag to non-zero, @extended may contain GPSTATUS error code ($GPID_ERR*).Retourne False et @error <> 0, @extended contient le code erreur GPSTATUS($GPID_ERR*).

#include <GDIPlus.au3>
#include <ScreenCapture.au3>
#include <WinAPI.au3>

Example()

Func Example()
    Local $hBitmap1, $hBitmap2, $hImage1, $hImage2, $hGraphic

    ; Initialize GDI+ libraryInitialise GDI+ Librairie
    _GDIPlus_Startup()

    ; Capture full screenCapture un plein écran
    $hBitmap1 = _ScreenCapture_Capture("")
    $hImage1 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap1)

    ; Capture screen regionCapture une région de l'écran
    $hBitmap2 = _ScreenCapture_Capture("", 0, 0, 400, 300)
    $hImage2 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap2)

    ; Draw one image in anotherDessine une image dans une autre
    $hGraphic = _GDIPlus_ImageGetGraphicsContext($hImage1)
    _GDIPlus_GraphicsDrawImageRect($hGraphic, $hImage2, 100, 100, 400, 300)

    ; Draw a frame around the inserted imageDessine un cadre autour de l'image insérée
    _GDIPlus_GraphicsDrawRect($hGraphic, 100, 100, 400, 300)

    ; Save resultant imageEnregistre l'image qui en résulte
    _GDIPlus_ImageSaveToFile($hImage1, @MyDocumentsDir & "\GDIPlus_Image.jpg")

    ; Clean up resourcesLibère les ressources
    _GDIPlus_ImageDispose($hImage1)
    _GDIPlus_ImageDispose($hImage2)
    _WinAPI_DeleteObject($hBitmap1)
    _WinAPI_DeleteObject($hBitmap2)

    ; Shut down GDI+ libraryFerme la librairie GDI+
    _GDIPlus_Shutdown()

    ShellExecute(@MyDocumentsDir & "\GDIPlus_Image.jpg")
EndFunc   ;==>Example

#include <GUIConstantsEx.au3>
#include <ScreenCapture.au3>

Global $g_hGUI, $g_hGfxCtxt, $g_hBitmap, $g_hBMP, $g_hGraphics

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ;initialize GDI+Initialize GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor format RRGGBB$iBGColor est de la forme RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;create a test GUIcréer une interface de test
    GUISetBkColor($iBgColor, $g_hGUI) ;set GUI background colorDéfinit la couleur de fond de la GUI
    GUISetState(@SW_SHOW)

    ;create buffered graphics frame set for smoother gfx object movementsCrée un buffer graphique pour lisser les objets gfx en mouvement
    $g_hGraphics = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;create a graphics object from a window handleCrée un objet graphique à partir du handle de fenêtre
    $g_hBitmap = _GDIPlus_BitmapCreateFromGraphics($iWidth, $iHeight, $g_hGraphics)
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBitmap)
    Local Const $iW = 300, $iH = 300
    Local $hHBmp = _ScreenCapture_Capture("", 0, @DesktopHeight - $iH, $iW, @DesktopHeight) ;create a GDI bitmap by capturing an area on desktopCrée un bitmap GDI en capturant une zone du bureau
    $g_hBMP = _GDIPlus_BitmapCreateFromHBITMAP($hHBmp) ;convert GDI to GDI+ bitmapCrée un bitmap GDI en capturant une région du bureau
    _WinAPI_DeleteObject($hHBmp) ;release GDI bitmap resource because not needed anymoreLibére la ressource du bitmap GDI car n'est plus nécessaire
    Local $iVectorX = Random(1.5, 2.5), $iVectorY = Random(1.5, 2.5) ;define x and y vectorDéfinit les composantes du vecteur direction
    Local $iX = 0.0, $iY = 0.0 ;define start coordinateDéfinit les coordonnées de départ

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ;clear bitmap for repaintEfface bitmap pour le repeindre
        _GDIPlus_GraphicsDrawImageRect($g_hGfxCtxt, $g_hBMP, $iX, $iY, $iW, $iH) ;draw bitmap to backbufferdessine le bitmap dans le backbuffer
        _GDIPlus_GraphicsDrawImageRect($g_hGraphics, $g_hBitmap, 0, 0, $iWidth, $iHeight) ;copy drawn bitmap to graphics handle (GUI)Copie le bitmap dessinée sur le 'graphics'
        $iX += $iVectorX ;add x vector to current x positionAjoute x du vecteur au x de la position courante
        $iY += $iVectorY ;add y vector to current y positionAjoute le y du vecteur au y de la position actuelle
        If $iX < 0 Or $iX > ($iWidth - $iW) Then $iVectorX *= -1 ;when x border is reached reverse x vectorQuand l'abscisse de la bordure est atteinte, l'abscisse du vecteur change de signe
        If $iY < 0 Or $iY > ($iHeight - $iH) Then $iVectorY *= -1 ;when y border is reached reverse y vectorQuand l'ordonnée de la bordure est atteinte, l'ordonnées du vecteur change de signe
    Until Not Sleep(20) ;sleep 20 ms to avoid high cpu usagePause de 20 ms pour éviter une utilisation élevée du processeur
EndFunc   ;==>Example

Func _Exit()
    ;cleanup GDI+ resourcesNettoie les ressources de GDI+
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_GraphicsDispose($g_hGraphics)
    _GDIPlus_BitmapDispose($g_hBitmap)
    _GDIPlus_BitmapDispose($g_hBMP)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
