_Crypt_EncryptFile

Encrypts a file with specified key and algorithmChiffre un fichier avec la clé spécifiée et l'algorithme

#include <Crypt.au3>
_Crypt_EncryptFile ( $sSourceFile, $sDestinationFile, $vCryptKey, $iAlgID )

ParametersParamètres

$sSourceFile File to processFichier à traiter
$sDestinationFile File to save the processed fileFichier où sera enregistré le fichier traité
$vCryptKey Password or handle to a key if the CALG_USERKEY flag is specifiedMot de passe ou descripteur (handle) d'une clé si l'indicateur CALG_USERKEY est spécifié
$iAlgID The algorithm to useL'algorithme à utiliser

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False and sets the @error flag to non-zero.False et met @error <> 0.
@error:@error : 10+ - Failed to create key10+ - Impossible de créer la clé
2 - Failed to open source file2 - Impossible d'ouvrir le fichier source
3 - Failed to open destination file3 - Impossible d'ouvrir le fichier de destination
400+ - Failed to encrypt final piece400+ - Impossible de chiffrer la dernière partie
500+ - Failed to encrypt piece500+ - Impossible de chiffrer la partie

RemarksRemarques

The the output file can be larger than the input file depending on the algorithm.Selon l'algorithme, le fichier de sortie peut être plus grand que le fichier d'entrée.

ExampleExemple

#include <ComboConstants.au3>
#include <Crypt.au3>
#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <StringConstants.au3>

Example()

Func Example()
    Local $iAlgorithm = $CALG_RC4

    Local $hGUI = GUICreate("File EncrypterChiffreur de fichiers", 425, 100)
    Local $idSourceInput = GUICtrlCreateInput("", 5, 5, 200, 20)
    Local $idSourceBrowse = GUICtrlCreateButton("...", 210, 5, 35, 20)

    Local $idDestinationInput = GUICtrlCreateInput("", 5, 30, 200, 20)
    Local $idDestinationBrowse = GUICtrlCreateButton("...", 210, 30, 35, 20)

    GUICtrlCreateLabel("Password:Mot de passe :", 5, 60, 200, 20)
    Local $idPasswordInput = GUICtrlCreateInput("", 5, 75, 200, 20)

    Local $idCombo = GUICtrlCreateCombo("", 210, 75, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "3DES|AES (128bit)|AES (192bit)|AES (256bit)|DES|RC2|RC4", "RC4")
    Local $idEncrypt = GUICtrlCreateButton("EncryptChiffrer", 355, 70, 65, 25)
    GUISetState(@SW_SHOW, $hGUI)

    Local $sDestinationRead = "", $sFilePath = "", $sPasswordRead = "", $sSourceRead = ""
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

            Case $idSourceBrowse
                $sFilePath = FileOpenDialog("Select a file to encrypt.Sélectionner un fichier à chiffrer.", "", "All files (*.*)Tous les fichiers (*.*)") ; Select a file to encrypt.Sélectionne un fichier à chiffrer.
                If @error Then
                    ContinueLoop
                EndIf
                GUICtrlSetData($idSourceInput, $sFilePath) ; Set the inputbox with the filepath.Définit l'inputbox avec le chemin du fichier.

            Case $idDestinationBrowse
                $sFilePath = FileSaveDialog("Save the file as ...Enregistrer le fichier sous...", "", "All files (*.*)Tous les fichiers (*.*)") ; Select a file to save the encrypted data to.Sélectionne un fichier pour enregistrer les données chiffrées.
                If @error Then
                    ContinueLoop
                EndIf
                GUICtrlSetData($idDestinationInput, $sFilePath) ; Set the inputbox with the filepath.Définit l'inputbox avec le chemin du fichier.

            Case $idCombo ; Check when the combobox is selected and retrieve the correct algorithm.Vérifie si le combobox est sélectionné et récupère le bon algorithme.
                Switch GUICtrlRead($idCombo) ; Read the combobox selection.Lit la sélection du combobox.
                    Case "3DES"
                        $iAlgorithm = $CALG_3DES

                    Case "AES (128bit)"
                        $iAlgorithm = $CALG_AES_128

                    Case "AES (192bit)"
                        $iAlgorithm = $CALG_AES_192

                    Case "AES (256bit)"
                        $iAlgorithm = $CALG_AES_256

                    Case "DES"
                        $iAlgorithm = $CALG_DES

                    Case "RC2"
                        $iAlgorithm = $CALG_RC2

                    Case "RC4"
                        $iAlgorithm = $CALG_RC4

                EndSwitch

            Case $idEncrypt
                $sSourceRead = GUICtrlRead($idSourceInput) ; Read the source filepath input.Lit le chemin du fichier source dans l'input.
                $sDestinationRead = GUICtrlRead($idDestinationInput) ; Read the destination filepath input.Lit le chemin du fichier de destination dans l'input.
                $sPasswordRead = GUICtrlRead($idPasswordInput) ; Read the password input.Lit le mot de passe dans l'input.
                If StringStripWS($sSourceRead, $STR_STRIPALL) <> "" And StringStripWS($sDestinationRead, $STR_STRIPALL) <> "" And StringStripWS($sPasswordRead, $STR_STRIPALL) <> "" And FileExists($sSourceRead) Then ; Check there is a file available to encrypt and a password has been set.Vérifie que le fichier à chiffrer existe et qu'un mot de passe a été défini.
                    If _Crypt_EncryptFile($sSourceRead, $sDestinationRead, $sPasswordRead, $iAlgorithm) Then ; Encrypt the file.Chiffre le fichier.
                        MsgBox($MB_SYSTEMMODAL, "SuccessSuccès", "Operation succeeded.Opération réussie.")
                    Else
                        Switch @error
                            Case 1
                                MsgBox($MB_SYSTEMMODAL, "ErrorErreur", "Failed to create the key.Impossible de créer la clé.")
                            Case 2
                                MsgBox($MB_SYSTEMMODAL, "ErrorErreur", "Couldn't open the source file.Impossible d'ouvrir le fichier source.")
                            Case 3
                                MsgBox($MB_SYSTEMMODAL, "ErrorErreur", "Couldn't open the destination file.Impossible d'ouvrir le fichier de destination.")
                            Case 4 Or 5
                                MsgBox($MB_SYSTEMMODAL, "ErrorErreur", "Encryption error.Erreur de chiffrement.")
                        EndSwitch
                    EndIf
                Else
                    MsgBox($MB_SYSTEMMODAL, "ErrorErreur", "Please ensure the relevant information has been entered correctly.Veuillez vous assurer que l'information appropriée a été entrée correctement.")
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Delete the previous GUI and all controls.Supprime la GUI précédente et tous ses contrôles.
EndFunc   ;==>Example
