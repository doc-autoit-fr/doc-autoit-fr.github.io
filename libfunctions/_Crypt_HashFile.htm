#include <Crypt.au3>
_Crypt_HashFile ( $sFilePath, $iAlgID )

ParametersParamètres

$iAlgID Hash ID to useID de hashage à utiliser

Return ValueValeur de retour

Success:Succès : a hash of file.le code de hachage d'un fichier.
Failure:Échec : -1 and sets the @error flag to non-zero.-1 et met @error <> 0.
@error:@error : 1 - Failed to open file1 - Impossible d'ouvrir le fichier
10+ - Failed to hash final piece10+ - Impossible de hacher la dernière partie

ExampleExemple

#include <ComboConstants.au3>
#include <Crypt.au3>
#include <GUIConstantsEx.au3>
#include <StringConstants.au3>

Example()

Func Example()
    Local $iAlgorithm = $CALG_SHA1

    Local $hGUI = GUICreate("Hash FileHach Fichier", 425, 70)
    Local $idInput = GUICtrlCreateInput(@ScriptFullPath, 5, 5, 200, 20)
    Local $idBrowse = GUICtrlCreateButton("...", 210, 5, 35, 20)
    Local $idCombo = GUICtrlCreateCombo("", 250, 5, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "MD2|MD4|MD5|SHA1", "SHA1")
    Local $idCalculate = GUICtrlCreateButton("CalculateCalculer", 355, 40, 65, 25)
    Local $idHashLabel = GUICtrlCreateLabel("Hash DigestSynthèse de Hachage (hash digest)", 5, 50, 350, 25)
    GUISetState(@SW_SHOW, $hGUI)

    _Crypt_Startup() ; To optimize performance start the crypt library.Exécute la bibliothèque crypt pour optimiser les performances.

    Local $dHash = 0, _
            $sRead = ""
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

            Case $idBrowse
                Local $sFilePath = FileOpenDialog("Open a fileOuvrir un fichier", "", "All files (*.*)Tous les fichiers (*.*)") ; Select a file to find the hash.Sélectionne un fichier pour trouver le hachage.
                If @error Then
                    ContinueLoop
                EndIf
                GUICtrlSetData($idInput, $sFilePath) ; Set the inputbox with the filepath.Définit l'inputbox avec le chemin du fichier.
                GUICtrlSetData($idHashLabel, "Hash DigestSynthèse de Hachage (hash digest)") ; Reset the hash digest label.Réinitialise le nom de la synthèse de hachage.

            Case $idCombo ; Check when the combobox is selected and retrieve the correct algorithm.Vérifie si le combobox est sélectionné et récupère le bon algorithme.
                Switch GUICtrlRead($idCombo) ; Read the combobox selection.Lit la sélection du combobox.
                    Case "MD2"
                        $iAlgorithm = $CALG_MD2

                    Case "MD4"
                        $iAlgorithm = $CALG_MD4

                    Case "MD5"
                        $iAlgorithm = $CALG_MD5

                    Case "SHA1"
                        $iAlgorithm = $CALG_SHA1

                EndSwitch

            Case $idCalculate
                $sRead = GUICtrlRead($idInput)
                If StringStripWS($sRead, $STR_STRIPALL) <> "" And FileExists($sRead) Then ; Check there is a file available to find the hash digestVérifie que le fichier existe pour rechercher la synthèse de hachage.
                    $dHash = _Crypt_HashFile($sRead, $iAlgorithm) ; Create a hash of the file.Crée un hachage du fichier.
                    GUICtrlSetData($idHashLabel, $dHash) ; Set the hash digest label with the hash data.Définit le nom de la synthèse de hachage avec les données de hachage.
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Delete the previous GUI and all controls.Supprime la GUI précédente et tous ses contrôles.
    _Crypt_Shutdown() ; Shutdown the crypt library.Arrête la bibliothèque crypt.
EndFunc   ;==>Example
