_GDIPlus_LineBrushSetColors

Sets the starting color and ending color of a linear gradient brushDéfinit la couleur de début et la couleur de fin d'un pinceau à dégradé linéaire

#include <GDIPlus.au3>
_GDIPlus_LineBrushSetColors ( $hLineGradientBrush, $iARGBStart, $iARGBEnd )

ParametersParamètres

$hLineGradientBrush Pointer to a LinearGradientBrush objectHandle de l'objet LinearGradientBrush
$iARGBStart Alpha, Red, Green and Blue components of the starting colorComposantes Alpha, Rouge, Vert et Bleu de la couleur de début
$iARGBEnd Alpha, Red, Green and Blue components of the ending colorComposantes Alpha, Rouge, Vert et Bleu de la couleur de fin

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False and sets the @error flag to non-zero, @extended may contain GPSTATUS error code ($GPID_ERR*).Retourne False et @error <> 0, @extended contient le code erreur GPSTATUS($GPID_ERR*).

ExampleExemple

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    _GDIPlus_Startup() ;initialize GDI+Initialize GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor format RRGGBB$iBGColor est de la forme RRGGBB

    Local $hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;create a test GUIcréer une interface de test
    GUISetBkColor($iBgColor, $hGUI) ;set GUI background colorDéfinit la couleur de fond de la GUI
    GUISetState(@SW_SHOW)

    Local $hGraphics = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;create a graphics object from a window handleCrée un objet graphique à partir du handle de fenêtre
    _GDIPlus_GraphicsSetSmoothingMode($hGraphics, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;sets the graphics object rendering quality (antialiasing)Définit l'objet graphique en qualité de rendu antialiasing
    Local $hBrush = _GDIPlus_LineBrushCreate(0, 0, 280, 570, 0xFFFFFF00, 0xFF4020FF, 1) ;create linear gradient flipped brushCrée un pinceau à dégradé linéaire renversé

    Local $aBlends[5][2]
    $aBlends[0][0] = 4 ; Using 4 factors and positionUtilise 4 facteurs et 4 emplacements
    $aBlends[1][0] = 0 ; FactorFacteur
    $aBlends[1][1] = 0 ; PositionEmplacement
    $aBlends[2][0] = 1 ; The percentage of the blending is 100% (red to green is gradually completed in just 20% from the left)Le pourcentage du mélange est de 100% (le dégradé du rouge au vert est progressivement complété dans seulement 20% à partir de la gauche)
    $aBlends[2][1] = 0.2 ; The distance percentage from the left boundary of the brush is 20%Le pourcentage de la distance de la frontière gauche du pinceau est de 20%
    $aBlends[3][0] = 0 ; The percentage of the blending is 0% (green to red is gradually completed from 20% to 70% from the left)Le pourcentage de mélange est de 0% (le dégradé du vert au rouge passe progressivement de 20% à 70 % à partir la gauche)
    $aBlends[3][1] = 0.7 ; The distance percentage from the left boundary of the brush is 70%Le pourcentage de la distance de la frontière gauche du pinceau est de 70%
    $aBlends[4][0] = 1 ; The percentage of the blending is 100% (red to green is gradually completed from 70% to 100% from the left)Le pourcentage de mélange est de 100% (le dégradé du rouge au vert est progressivement complété de 70% à 100 % à partir de la gauche
    $aBlends[4][1] = 1 ; The distance percentage from the left boundary of the brush is 100%Le pourcentage de la distance de la frontière gauche du pinceau est à 100%

    _GDIPlus_LineBrushSetBlend($hBrush, $aBlends); set the linear gradient brush color positions and factorsDéfinit les emplacements de couleurs de pinceau de dégradé linéaire et les facteurs

    _GDIPlus_GraphicsFillEllipse($hGraphics, 100, 50, 380, 500, $hBrush) ;draw the eggDessine l'œuf

    Sleep(2000)

    _GDIPlus_GraphicsClear($hGraphics, 0xFF000000 + $iBgColor) ;clear bitmapEfface le bitmap
    _GDIPlus_LineBrushSetColors($hBrush, 0xFFFFFFFF, 0xFF000000) ;change the color values to greyscaleChange les valeurs des couleurs en niveaux de gris

    _GDIPlus_GraphicsFillEllipse($hGraphics, 100, 50, 380, 500, $hBrush) ;draw the egg againDessine de nouveau l'œuf

    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ;cleanup GDI+ resourcesNettoie les ressources de GDI+
    _GDIPlus_BrushDispose($hBrush)
    _GDIPlus_GraphicsDispose($hGraphics)
    _GDIPlus_Shutdown()
    GUIDelete($hGUI)
EndFunc   ;==>Example
