_GUICtrlTreeView_SetIcon

Set an item icon

#include <GuiTreeView.au3>
_GUICtrlTreeView_SetIcon ( $hWnd [, $hItem = 0 [, $sIconFile = "" [, $iIconID = 0 [, $iImageMode = 6]]]] )

ParametersParamètres

$hWnd Control ID/Handle to the control
$hItem [optional][optionnel] item ID/handle
$sIconFile [optional][optionnel] The file to extract the icon of
$iIconID [optional][optionnel] The iconID to extract of the file
$iImageMode [optional][optionnel] 2=normal image / 4=selected image to set

Return ValueValeur de retour

Success:Succès : True.
Failure:Échec : False.Retourne False.

ExampleExemple

#include <GUIConstantsEx.au3>
#include <GuiTreeView.au3>
#include <WindowsConstants.au3>

Example()

Func Example()
    Local $aidItem[10], $idTreeView
    Local $iStyle = BitOR($TVS_EDITLABELS, $TVS_HASBUTTONS, $TVS_HASLINES, $TVS_LINESATROOT, $TVS_DISABLEDRAGDROP, $TVS_SHOWSELALWAYS)

    GUICreate("TreeView Set Icon", 400, 300)

    $idTreeView = GUICtrlCreateTreeView(2, 2, 396, 268, $iStyle, $WS_EX_CLIENTEDGE)
    GUISetState(@SW_SHOW)

    _GUICtrlTreeView_BeginUpdate($idTreeView)
    For $x = 0 To 9
        $aidItem[$x] = GUICtrlCreateTreeViewItem(StringFormat("[%02d] New Item", $x), $idTreeView)
        _GUICtrlTreeView_SetIcon($idTreeView, $aidItem[$x], "shell32.dll", 3)
        For $y = 1 To 3
            GUICtrlCreateTreeViewItem(StringFormat("[%02d] New Child", $y), $aidItem[$x])
        Next
    Next
    _GUICtrlTreeView_EndUpdate($idTreeView)

    _GUICtrlTreeView_SetIcon($idTreeView, $aidItem[0], "shell32.dll", 4)

    ; Loop until the user exits.Boucle jusqu'à ce que l'utilisateur quitte.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example
